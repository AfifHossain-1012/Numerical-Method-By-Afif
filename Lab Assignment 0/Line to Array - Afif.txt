    string s;
    vector<int> arr;

    cout << "Enter a line of integers separated by spaces: ";
    getline(cin, s);

    for (int i = 0; i < s.length(); i++)
    {
        if (s[i] == ' ')
        {
            continue; //Skip spaces
        }

        int start = i;
        while (i<s.length() && s[i]!=' ')
        {
            i++;
        }

        string numStr = s.substr(start, i - start);
        int ans = stoi(numStr);
        arr.push_back(ans);
    }

    cout << "The integers in the array are: ";
    for (int i=0; i<arr.size(); i++)
    {
        cout << arr[i] << " ";
    }
    cout << endl;

    return 0;
}
